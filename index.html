'translateX(0)' : 'translateX(100%)'; document.body.style.overflow = isMobileMenuOpen ? 'hidden' : 'auto'; mobileMenuButton.setAttribute('aria-expanded', isMobileMenuOpen); }; const closeMobileMenu = () => { if(isMobileMenuOpen) { toggleMobileMenu(); } }; // Event Listeners // Desktop Menu if(megaMenu) { megaMenu.parentElement.addEventListener('mouseenter', showMegaMenu); megaMenu.parentElement.addEventListener('mouseleave', hideMegaMenu); megaMenu.addEventListener('mouseenter', showMegaMenu); megaMenu.addEventListener('mouseleave', hideMegaMenu); } // Mobile Menu mobileMenuButton.addEventListener('click', toggleMobileMenu); mobileMenuClose.addEventListener('click', toggleMobileMenu); // Touch Gestures for Mobile Menu let touchStartX = 0; mobileMenu.addEventListener('touchstart', e => { touchStartX = e.touches[0].clientX; }); mobileMenu.addEventListener('touchend', e => { const touchEndX = e.changedTouches[0].clientX; if(touchStartX - touchEndX > 50) { closeMobileMenu(); } }); // Click Outside Handling document.addEventListener('click', e => { if(isMobileMenuOpen && !mobileMenu.contains(e.target) && !mobileMenuButton.contains(e.target)) { closeMobileMenu(); } }); // Keyboard Navigation document.addEventListener('keydown', e => { if(e.key === 'Escape') closeMobileMenu(); }); // Responsive Resize Handling const resizeObserver = new ResizeObserver(entries => { const width = entries[0].contentRect.width; if(width > 768) { // Desktop closeMobileMenu(); megaMenu.style.cssText = ''; // Reset styles } else { // Mobile megaMenu.style.transform = ''; megaMenu.style.opacity = ''; megaMenu.style.pointerEvents = ''; } }); resizeObserver.observe(document.body); // Cleanup window.addEventListener('beforeunload', () => { resizeObserver.disconnect(); if(megaMenu) { megaMenu.parentElement.removeEventListener('mouseenter', showMegaMenu); megaMenu.parentElement.removeEventListener('mouseleave', hideMegaMenu); megaMenu.removeEventListener('mouseenter', showMegaMenu); megaMenu.removeEventListener('mouseleave', hideMegaMenu); } }); });
